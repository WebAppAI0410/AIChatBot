## 7. ノートモジュール（実装順ガイド・最新版）

### 7.1 機能詳細

- **タブ**: ファイル・ノート一覧画面（フォルダ階層構造）、新規ノート/フォルダFAB（下タブと重ならないように配置）
- **エディタ**: WebViewベースWYSIWYG（React Native WebView + TipTap v2/TOAST UI Editor）。Apple純正ノートのように、タップで編集モードに入り、入力は自動保存。
- **テキスト選択**: ノート本文の一部をテキスト選択すると、アクションバブル（`AIアシスト`ボタン）が表示され、選択部分のみを編集・整形できる。
- **AIアシスト**: 右上「スパークル」ボタン → チャットドロワー表示。「質問」モード（ノート内容に関するQ&Aのみ、編集不可）と「編集」モード（AIによるノート編集提案・差分適用可）を切替可能。部分選択時は選択部分のみを対象としたAI編集も可能。
- **ノート一覧AIアシスト**: ノート一覧画面右上「スパークル」ボタン → チャットドロワー表示。ノート全体の管理や複数ノートのタグ一括編集などが可能。
- **タグオーガナイザー**: Lite+のみ: メニュー → 「タグを整理」（タグ名変更機能含む）
- **ストレージ**: expo-sqliteを使用（v1ではクラウド同期なし）
- **オフライン動作**: AIアシスト以外の機能はオフラインで完全動作

### 7.3 実装手順

1. **基本データ構造とSQLiteセットアップ** - ノート・フォルダ・タグのスキーマ設計
2. **ノート一覧UI** - フォルダ階層表示、新規作成FAB（タブと重ならない位置に配置）、AIアシストドロワー
3. **ノートエディタUI** - Apple純正ノート風のタップ編集・自動保存、AIアシスト（質問/編集モード切替）
4. **テキスト選択AIアシスト** - 部分選択時のアクションバブル、選択範囲のAI編集
5. **タグ管理機能** - タグの追加・編集・削除、複数ノートへの一括タグ付け
6. **差分表示機能** - AIによるノート編集と変更確認

### 7.5 仕様補足
- ノート・タグ・フォルダの作成/編集/削除は全て自動保存
- FABはタブUIと重ならないように絶対配置

- AIアシストはOpenRouter API、プラン別クォータ・モデル適用
- ドロワー表示中は背面編集不可（編集再開時はドロワーを閉じる）

// Android用のSQLiteファイル名フォーマット修正

  if (db) return db;

    // WebでのSQLiteはサポート外

// 初期化処理
export const initDatabase = async (): Promise<void> => {
  const db = getDB();
  
  // トランザクションで実行
  return new Promise((resolve, reject) => {
    db.transaction(tx => {
      // フォルダテーブル作成
      tx.executeSql(
        `CREATE TABLE IF NOT EXISTS folders (
          id TEXT PRIMARY KEY,
          name TEXT NOT NULL,
          parent_id TEXT,
          created_at TEXT NOT NULL,
          updated_at TEXT NOT NULL,
          FOREIGN KEY (parent_id) REFERENCES folders (id) ON DELETE CASCADE
        );`,
        [],
        () => {},
        (_, error): boolean => {
          reject(error);
          return false;
        }
      );
      
      // ノートテーブル作成
      tx.executeSql(
        `CREATE TABLE IF NOT EXISTS notes (
          id TEXT PRIMARY KEY,
          title TEXT NOT NULL,
          content TEXT NOT NULL,
          folder_id TEXT,
          created_at TEXT NOT NULL,
          updated_at TEXT NOT NULL,
          FOREIGN KEY (folder_id) REFERENCES folders (id) ON DELETE CASCADE
        );`,
        [],
        () => {},
        (_, error): boolean => {
          reject(error);
          return false;
        }
      );
      
      // タグテーブル作成
      tx.executeSql(
        `CREATE TABLE IF NOT EXISTS tags (
          id TEXT PRIMARY KEY,
          name TEXT NOT NULL UNIQUE
        );`,
        [],
        () => {},
        (_, error): boolean => {
          reject(error);
          return false;
        }
      );
      
      // ノートとタグの関連テーブル作成
      tx.executeSql(
        `CREATE TABLE IF NOT EXISTS note_tags (
          note_id TEXT NOT NULL,
          tag_id TEXT NOT NULL,
          PRIMARY KEY (note_id, tag_id),
          FOREIGN KEY (note_id) REFERENCES notes (id) ON DELETE CASCADE,
          FOREIGN KEY (tag_id) REFERENCES tags (id) ON DELETE CASCADE
        );`,
        [],
        () => {},
        (_, error): boolean => {
          reject(error);
          return false;
        }
      );
    }, 
    error => {
      reject(error);
    },
    () => {
      resolve();
    });
  });
};

// フォルダ操作
export const createFolder = async (folder: Omit<Folder, 'id' | 'created_at' | 'updated_at'>): Promise<Folder> => {
  const db = getDB();
  const id = `folder_${Date.now()}`;
  const now = new Date().toISOString();
  
  return new Promise((resolve, reject) => {
    db.transaction(tx => {
      tx.executeSql(
        `INSERT INTO folders (id, name, parent_id, created_at, updated_at)
         VALUES (?, ?, ?, ?, ?)`,
        [id, folder.name, folder.parent_id, now, now],
        (_, result) => {
          resolve({
            id,
            name: folder.name,
            parent_id: folder.parent_id,
            created_at: now,
            updated_at: now
          });
        },
        (_, error): boolean => {
          reject(error);
          return false;
        }
      );
    });
  });
};

### 7.8 WebViewベースエディタの基本コンポーネント

```typescript
// components/WebViewEditor.tsx
import React, { useRef, useEffect, useState, useCallback } from 'react';
import { View, StyleSheet, ActivityIndicator } from 'react-native';
import { WebView } from 'react-native-webview';
import * as FileSystem from 'expo-file-system';

const editorHTML = require('../assets/editor/editor.html');

type WebViewEditorProps = {
  initialContent: string;
  onChange: (content: string) => void;
  readOnly?: boolean;
  onTextSelect?: (selectedText: string) => void; // テキスト選択イベント
};

export const WebViewEditor: React.FC<WebViewEditorProps> = ({
  initialContent,
  onChange,
  readOnly = false,
  onTextSelect,
}, ref) => {
  const webViewRef = useRef<WebView>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [editorReady, setEditorReady] = useState(false);
  const [html, setHtml] = useState<string>('');

  // エディタHTMLの読み込み
  useEffect(() => {
    const loadEditorHTML = async () => {
      try {
        // アセットからHTMLを読み込む（実際の構成に合わせて調整してください）
        const fileUri = FileSystem.documentDirectory + 'editor.html';
        await FileSystem.copyAsync({
          from: editorHTML,
          to: fileUri
        });
        
        const htmlContent = await FileSystem.readAsStringAsync(fileUri);
        setHtml(htmlContent);
      } catch (error) {
        console.error('Failed to load editor HTML:', error);
      }
    };
    
    loadEditorHTML();
  }, []);

  // エディタの準備完了後、初期コンテンツを設定
  useEffect(() => {
    if (editorReady && webViewRef.current) {
      webViewRef.current.injectJavaScript(`
        editor.setContent(\`${initialContent.replace(/`/g, '\\`')}\`);
        ${readOnly ? 'editor.setReadOnly(true);' : ''}
        true;
      `);
    }
  }, [editorReady, initialContent, readOnly]);

  // WebViewからのメッセージを処理
  const handleMessage = useCallback((event) => {
    try {
      const data = JSON.parse(event.nativeEvent.data);
      
      switch (data.type) {
        case 'ready':
          setEditorReady(true);
          setIsLoading(false);
          break;
        
        case 'change':
          onChange(data.content);
          break;

        case 'textSelect':
          if (onTextSelect && data.selectedText) {
            onTextSelect(data.selectedText);
          }
          break;
          
        case 'error':
          console.error('Editor error:', data.message);
          break;
      }
    } catch (error) {
      console.error('Failed to parse WebView message:', error);
    }
  }, [onChange, onTextSelect]);

  // 選択範囲テキストの置換（AIアシスト後の適用用）
  const replaceSelectedText = useCallback((newText: string) => {
    if (webViewRef.current && editorReady) {
      webViewRef.current.injectJavaScript(`
        editor.replaceSelectedText(\`${newText.replace(/`/g, '\\`')}\`);
        true;
      `);
    }
  }, [editorReady]);

  // コンポーネントインターフェースを拡張（ref経由で呼び出せるように）
  React.useImperativeHandle(
    ref,
    () => ({
      replaceSelectedText,
    }),
    [replaceSelectedText]
  );

  if (!html) {
    return (
      <View style={styles.loadingContainer}>
        <ActivityIndicator size="large" />
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <WebView
        ref={webViewRef}
        source={{ html }}
        style={styles.webView}
        onMessage={handleMessage}
        originWhitelist={['*']}
        javaScriptEnabled={true}
        domStorageEnabled={true}
        startInLoadingState={true}
        onLoadEnd={() => setIsLoading(false)}
        renderLoading={() => (
          <View style={styles.loadingContainer}>
            <ActivityIndicator size="large" />
          </View>
        )}
      />
      {isLoading && (
        <View style={styles.overlay}>

export const NoteAIAssist = ({
  noteContent,
  onClose,
  onApplySuggestion,

  noteContent: string;
  onClose: () => void;
  onApplySuggestion: (suggestion: string) => void;

  const { t } = useTranslation();
  const { plan, dailyModelQuotasCount, incrementModelUsageCount } = useUserStore();

  const [messages, setMessages] = useState<Message[]>([]);
  const [input, setInput] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const getAIModel = useCallback(() => {
    switch (plan) {
      case 'premium':
        return 'anthropic/claude-3-sonnet-20240229';
      case 'lite':
        return 'openai/gpt-4o';
      default:
        return 'openai/gpt-4o-mini';
    }

  // 残りのAIアシストクォータを確認
  const getRemainingQuota = useCallback(() => {
    const quotaLimits = {
      premium: 50,
      lite: 20,
      free: 5
    };
    
    const limit = quotaLimits[plan] || quotaLimits.free;
    const used = dailyModelQuotasCount['note_ai_assist'] || 0;
    
    return Math.max(0, limit - used);
  }, [plan, dailyModelQuotasCount]);

  // システムプロンプトの生成
  const getSystemPrompt = useCallback(() => {
    return `あなたはノート編集のプロフェッショナルアシスタントです。
以下のノート内容について、ユーザーの質問に答えたり、編集の提案をしたりしてください。
編集提案をする場合は、元の文脈や意図を尊重しながら、改善点を具体的に示してください。

===== ノート内容 =====
${noteContent || 'まだノートに内容がありません。'}
====================`;
  }, [noteContent]);

  // メッセージ送信処理
  const sendMessage = useCallback(async () => {
    if (!input.trim() || isLoading || getRemainingQuota() <= 0) return;
    
    setError(null);

    const userMessage: Message = {
      id: `msg_${Date.now()}`,
      role: 'user',
      content: input,
    };

    setMessages(prev => [...prev, userMessage]);
    setInput('');
    setIsLoading(true);

    try {
      // 初めてのメッセージ送信時にはシステムメッセージを追加
      const allMessages = messages.length === 0 
        ? [{ role: 'system', content: getSystemPrompt() }, userMessage]
        : [...messages, userMessage];
        
      const response = await generateAIAssist({
        messages: allMessages.map(m => ({ role: m.role, content: m.content })),
        model: getAIModel(),
      });

      const assistantMessage: Message = {
        id: `msg_${Date.now() + 1}`,
        role: 'assistant',
        content: response,
      };

      setMessages(prev => [...prev, assistantMessage]);
      incrementModelUsageCount('note_ai_assist');
    } catch (error) {
      console.error('AI Assist error:', error);
      setError(typeof error === 'string' ? error : 'AIアシストの呼び出しに失敗しました');
    } finally {
      setIsLoading(false);
    }
  }, [input, isLoading, messages, getSystemPrompt, getAIModel, incrementModelUsageCount]);

  // 提案をノートに適用
  const handleApplySuggestion = useCallback((content: string) => {
    onApplySuggestion(content);

  const requestEditSuggestion = useCallback(() => {
    setInput(t('suggest_improvements_to_this_note'));

  }, [t, sendMessage]);

        return 4000;
      case 'lite':
        return 2000;

        return 1000;
    }
  }, [plan]);

    <View style={styles.container}>
      <XStack py="$2" px="$3" alignItems="center" justifyContent="space-between">
        <Text fontSize="$4" fontWeight="bold">
          {t('ai_assist')} ({getRemainingQuota()}/{
            plan === 'premium' ? 50 : plan === 'lite' ? 20 : 5
          })
        </Text>
        <Button icon={X} variant="ghost" circular onPress={onClose} />
      </XStack>

      <ScrollView style={styles.messagesContainer}>
        {messages.length === 0 ? (
          <YStack p="$4" alignItems="center" space="$4">
            <Text fontSize="$3" textAlign="center" color={theme.colors.textSecondary}>
              {t('ai_assist_instruction')}
            </Text>
            <Button
              icon={Wand}
              onPress={requestEditSuggestion}
              disabled={getRemainingQuota() <= 0}
            >
              {t('suggest_improvements')}
            </Button>
          </YStack>
        ) : (
          <YStack space="$3" p="$3">
            {messages.filter(m => m.role !== 'system').map(message => (
              <View
                key={message.id}
                style={[
                  styles.messageBubble,
                  message.role === 'user' ? styles.userBubble : styles.assistantBubble
                ]}
              >
                <Text selectable>{message.content}</Text>
                {message.role === 'assistant' && (
                  <Button
                    size="$2"
                    mt="$2"
                    variant="outlined"
                    onPress={() => handleApplySuggestion(message.content)}
                  >
                    {t('apply_to_note')}

                )}

/**

export const generateAIAssist = async ({
  messages,
  model,
}: {
  messages: { role: string; content: string }[];
  model: string;
}): Promise<string> => {
  try {
    // OpenRouter APIを使用
    const API_BASE_URL = 'https://openrouter.ai/api/v1';
    
    // APIリクエスト
    const response = await fetch(`${API_BASE_URL}/chat/completions`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${await getApiKey('openrouter')}`,
        'HTTP-Referer': 'https://your-app-domain.com', // アプリのドメインに変更
        'X-Title': 'AI Chat Bot',
      },
      body: JSON.stringify({
        model,
        messages,
        max_tokens: model.includes('claude') ? 4000 : (model.includes('mini') ? 1000 : 2000),
        temperature: 0.7,
      }),
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.error?.message || 'AIアシストの生成に失敗しました');
    }

    const data = await response.json();
    return data.choices[0].message.content;
  } catch (error) {
    console.error('Failed to generate AI assist:', error);
    throw error;
  }
};

          window.ReactNativeWebView.postMessage(JSON.stringify({
            type: 'change',
            content
          }));
        },
        onSelectionUpdate: ({ editor }) => {
          handleTextSelection(editor);
        }
      });
      
      // 準備完了を通知
      window.ReactNativeWebView.postMessage(JSON.stringify({
        type: 'ready'
      }));
    });
    
    // テキスト選択処理
    function handleTextSelection(editor) {
      // 現在の選択範囲を取得
      const { from, to } = editor.state.selection;
      
      // 選択がなければアクションバブルを削除
      if (from === to) {
        hideActionBubble();
        selectedText = '';
        return;
      }
      
      // 選択テキストを取得
      selectedText = editor.state.doc.textBetween(from, to, ' ');
      
      // 選択範囲が変わったらアクションバブルを表示
      if (selectedText) {
        showActionBubble();
      }
    }
    
    // アクションバブルを表示
    function showActionBubble() {
      // 既存のバブルを削除
      hideActionBubble();
      
      // バブルを生成
      actionBubble = document.createElement('div');
      actionBubble.className = 'action-bubble';
      actionBubble.innerText = 'AIアシスト';
      actionBubble.addEventListener('click', () => {
        // React Nativeへ選択テキストを通知
        window.ReactNativeWebView.postMessage(JSON.stringify({
          type: 'textSelect',
          selectedText
        }));
        
        // バブルを削除
        hideActionBubble();
      });
      
      // 配置位置計算（選択範囲の上に表示）
      const selection = window.getSelection();
      if (selection.rangeCount > 0) {
        const range = selection.getRangeAt(0);
        const rect = range.getBoundingClientRect();
        
        actionBubble.style.top = `${rect.top - 10}px`;
        actionBubble.style.left = `${(rect.left + rect.right) / 2}px`;
        
        document.body.appendChild(actionBubble);
      }
    }
    
    // アクションバブルを削除
    function hideActionBubble() {
      if (actionBubble) {
        actionBubble.remove();
        actionBubble = null;
      }
    }
    
    // 選択テキストを新しいテキストで置換
    window.replaceSelectedText = (newText) => {
      if (editor && selectedText) {
        // 現在の選択範囲のテキストを置換
        const { from, to } = editor.state.selection;
        editor.chain()
          .focus()
          .deleteRange({ from, to })
          .insertContent(newText)
          .run();
          
        // コンテンツ変更を通知
        window.ReactNativeWebView.postMessage(JSON.stringify({
          type: 'change',
          content: editor.getHTML()
        }));
      }
    };
    
    // React Nativeからの操作を処理
    window.setContent = (content) => {
      if (editor) {
        editor.commands.setContent(content);
      }
    };
    
    window.setReadOnly = (readonly) => {
      if (editor) {
        editor.setEditable(!readonly);
      }
    };
  </script>
</body>
</html>
```

// components/note/TagSelector.tsx
import React, { useState, useCallback, useEffect } from 'react';
import { View, StyleSheet, TextInput, FlatList, ActivityIndicator } from 'react-native';
import { Text, YStack, XStack, Button, Checkbox } from 'tamagui';
import { X, Plus, Trash, Edit } from '@tamagui/lucide-icons';
import { useTranslation } from 'react-i18next';
import { useNoteStore } from '../../store/noteStore';
import { theme } from '../../ui/theme';

export const TagSelector = ({
  selectedTags,
  onTagsChange,
  onClose,
}: {
  selectedTags: any[];
  onTagsChange: (tags: any[]) => void;
  onClose: () => void;
}) => {
  const { t } = useTranslation();
  const { 
    allTags, 
    addTag, 
    updateTag,
    removeTag 
  } = useNoteStore();

  const [newTag, setNewTag] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [allTagsList, setAllTagsList] = useState<any[]>([]);
  const [editingTag, setEditingTag] = useState<{id: string, name: string} | null>(null);
  const [editedTagName, setEditedTagName] = useState('');

  // タグ一覧の読み込み
  useEffect(() => {
    const loadTags = async () => {
      setIsLoading(true);
      try {
        const tags = await allTags();
        setAllTagsList(tags);
      } catch (error) {
        console.error('Failed to load tags:', error);
      } finally {
        setIsLoading(false);
      }
    };
    
    loadTags();
  }, [allTags]);

  // タグ選択処理
  const toggleTag = useCallback((tag) => {
    const isSelected = selectedTags.some(t => t.id === tag.id);
    
    if (isSelected) {
      onTagsChange(selectedTags.filter(t => t.id !== tag.id));
    } else {
      onTagsChange([...selectedTags, tag]);
    }
  }, [selectedTags, onTagsChange]);

  // 新規タグ追加
  const handleAddTag = useCallback(async () => {
    if (!newTag.trim() || allTagsList.some(t => t.name.toLowerCase() === newTag.trim().toLowerCase())) return;
    
    setIsLoading(true);
    try {
      const newTagId = await addTag(newTag.trim());
      const newTagObj = { id: newTagId, name: newTag.trim() };
      
      setAllTagsList(prev => [...prev, newTagObj]);
      onTagsChange([...selectedTags, newTagObj]);
    setNewTag('');
    } catch (error) {
      console.error('Failed to add tag:', error);
    } finally {
      setIsLoading(false);
    }
  }, [newTag, allTagsList, addTag, selectedTags, onTagsChange]);

  // タグ名編集モード開始
  const startEditingTag = useCallback((tag) => {
    setEditingTag(tag);
    setEditedTagName(tag.name);
  }, []);

  // タグ名更新
  const handleUpdateTag = useCallback(async () => {
    if (!editingTag || !editedTagName.trim() || editedTagName === editingTag.name || 
        allTagsList.some(t => t.id !== editingTag.id && t.name.toLowerCase() === editedTagName.trim().toLowerCase())) {
      setEditingTag(null);
      return;
    }
    
    setIsLoading(true);
    try {
      await updateTag(editingTag.id, editedTagName.trim());
      
      // タグリストの更新
      setAllTagsList(prev => prev.map(t => 
        t.id === editingTag.id ? { ...t, name: editedTagName.trim() } : t
      ));
      
      // 選択中タグの更新
      onTagsChange(selectedTags.map(t => 
        t.id === editingTag.id ? { ...t, name: editedTagName.trim() } : t
      ));
      
      setEditingTag(null);
    } catch (error) {
      console.error('Failed to update tag:', error);
    } finally {
      setIsLoading(false);
    }
  }, [editingTag, editedTagName, allTagsList, updateTag, selectedTags, onTagsChange]);

  // タグ削除
  const handleRemoveTag = useCallback(async (tagId) => {
    setIsLoading(true);
    try {
      await removeTag(tagId);
      
      setAllTagsList(prev => prev.filter(t => t.id !== tagId));
      onTagsChange(selectedTags.filter(t => t.id !== tagId));
    } catch (error) {
      console.error('Failed to remove tag:', error);
    } finally {
      setIsLoading(false);
    }
  }, [removeTag, selectedTags, onTagsChange]);

  // 編集キャンセル
  const cancelEditing = useCallback(() => {
    setEditingTag(null);
  }, []);

  return (
    <View style={styles.container}>
      <XStack py="$2" px="$3" alignItems="center" justifyContent="space-between">
        <Text fontSize="$4" fontWeight="bold">{t('manage_tags')}</Text>
        <Button icon={X} variant="ghost" circular onPress={onClose} />
      </XStack>

      <XStack px="$3" py="$2" space="$2">
        <TextInput
          style={styles.input}
          value={newTag}
          onChangeText={setNewTag}
          placeholder={t('new_tag')}
          placeholderTextColor={theme.colors.textSecondary}
        />
        <Button
          icon={Plus}
          variant="primary"
          onPress={handleAddTag}
          disabled={!newTag.trim() || allTagsList.some(t => t.name.toLowerCase() === newTag.trim().toLowerCase())}
        />
      </XStack>

      {isLoading && allTagsList.length === 0 ? (
        <YStack p="$4" alignItems="center">
          <ActivityIndicator color={theme.colors.primary} />
        </YStack>
      ) : (
      <FlatList
          data={allTagsList}
          keyExtractor={item => item.id}
        renderItem={({ item }) => (
          <XStack px="$3" py="$2" alignItems="center" justifyContent="space-between">
              {editingTag?.id === item.id ? (
                // タグ編集モード
                <XStack space="$2" alignItems="center" flex={1}>
                  <TextInput
                    style={styles.editInput}
                    value={editedTagName}
                    onChangeText={setEditedTagName}
                    autoFocus
                  />
                  <Button
                    icon={Check}
                    size="$2"
                    variant="ghost"
                    onPress={handleUpdateTag}
                  />
                  <Button
                    icon={X}
                    size="$2"
                    variant="ghost"
                    onPress={cancelEditing}
                  />
                </XStack>
              ) : (
                // 通常モード
            <XStack space="$2" alignItems="center" flex={1}>
              <Checkbox
                    checked={selectedTags.some(t => t.id === item.id)}
                onCheckedChange={() => toggleTag(item)}
              />
                  <Text fontSize="$3">#{item.name}</Text>
            </XStack>
              )}
              
              <XStack space="$1">
                {editingTag?.id !== item.id && (
                  <Button
                    icon={Edit}
                    variant="ghost"
                    size="$2"
                    onPress={() => startEditingTag(item)}
                  />
                )}
            <Button
              icon={Trash}
              variant="ghost"
              size="$2"
                  onPress={() => handleRemoveTag(item.id)}
            />
              </XStack>
          </XStack>
        )}
        ListEmptyComponent={
          <YStack p="$4" alignItems="center">
            <Text color={theme.colors.textSecondary}>{t('no_tags')}</Text>
          </YStack>
        }
      />
      )}
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    position: 'absolute',
    top: 0,
    left: 0,
    right: 0,
    bottom: 0,
    backgroundColor: theme.colors.background,
    zIndex: 1000,
  },
  input: {
    flex: 1,
    backgroundColor: theme.colors.inputBackground,
    borderRadius: 8,
    paddingHorizontal: 16,
    paddingVertical: 8,
    color: theme.colors.text,
  },
  editInput: {
    flex: 1,
    backgroundColor: theme.colors.inputBackground,
    borderRadius: 8,
    paddingHorizontal: 12,
    paddingVertical: 6,
    color: theme.colors.text,
  }
});
