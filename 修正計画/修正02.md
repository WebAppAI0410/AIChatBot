### 1.3 Auto Router機能

#### 概要
OpenRouterのAuto Router機能を実装し、各プランでデフォルトのモデルとして「Auto」オプションを提供します。この機能により、サーバー負荷やモデル可用性に応じて最適なAIが自動選択されるようになります。

#### プラン別Auto Router動作
| プラン | 自動選択範囲 | 制限 | 表示 |
|--------|------------|------|------|

| **Lite** | 中コストモデルまで | Tier 0-1 | 現在使用中のモデル名を小さく表示 |
| **Premium** | すべてのモデル | Tier 0-2 | 現在使用中のモデル名を小さく表示 |

  ```typescript
  // services/api.ts
  export const getChatCompletion = async (messages: Message[], modelId: string) => {
    const userState = useStore.getState();
    const selectedModel = MODELS[modelId];

    // クォータチェック
    const quotaStatus = userState.checkQuotaStatus('tokens');

    // モデル選択ロジック
    let effectiveModel = modelId;
    let useAutoRouter = false;

    // Auto Routerの処理
    if (selectedModel.isAuto) {
      useAutoRouter = true;
      // プラン別のAuto Router制限を適用
      const userPlan = userState.plan;
      switch (userPlan) {
        case 'free':
          // Freeプランは低コストモデルのみ自動選択
          effectiveModel = 'auto'; // OpenRouterに低コストモデルのみを選択させる
          break;
        case 'lite':
          // Liteプランは中コストモデルまで自動選択
          effectiveModel = 'auto'; // OpenRouterに中コストモデルまでを選択させる
          break;
        case 'premium':
          // Premiumプランは高コストモデルも含めて自動選択
          effectiveModel = 'auto'; // OpenRouterに全モデルから選択させる
          break;
      }
    } else if (quotaStatus === 'exceeded') {
      // クォータ超過時のフォールバックロジック
      const userPlan = userState.plan;
      const availableTiers = PLANS[userPlan.toUpperCase()].modelTiers;

      // 現在のモデルTierより低いTierを探す
      const lowerTiers = availableTiers.filter(tier => tier < selectedModel.tier);

      if (lowerTiers.length > 0) {
        // 最も高いTierのモデルを選択
        const fallbackTier = Math.max(...lowerTiers);
        // 該当Tierのモデルから一つ選択
        const fallbackModels = Object.entries(MODELS)
          .filter(([_, model]) => model.tier === fallbackTier)
          .map(([id, _]) => id);

        if (fallbackModels.length > 0) {
          effectiveModel = fallbackModels[0]; // 最初のモデルを使用
          // フォールバック通知
          showToast(`クォータ超過のため ${MODELS[effectiveModel].name} にフォールバックしました`);
        }
      } else {
        // フォールバック不可の場合
        throw new Error('月間クォータを超過しました。プランをアップグレードするか、翌月まで待ってください。');
      }
    }

    // APIリクエスト実行
    // ...
  };
  ```

  ```typescript
  // services/subscription.ts
  import * as StoreKit from 'expo-store-kit';
  import * as SecureStore from 'expo-secure-store';
  import { supabase } from './supabase';
  import { PLANS } from '../constants/plans';
  import useStore from '../store';

  // レシート検証
  export const verifyReceipt = async (receipt: string, platform: 'ios' | 'android') => {
    try {
      // Supabase Edge Functionでレシート検証
      const { data, error } = await supabase.functions.invoke('verify-receipt', {
        body: { receipt, platform },
      });

      if (error) throw error;

      if (data.valid) {
        // 有効なレシート
        const { productId, purchaseDate, expiryDate } = data;

        // プラン情報の更新
        let planId = 'free';

        // productIdからプランを特定
        Object.entries(PLANS).forEach(([key, plan]) => {
          if (plan.periods) {
            Object.values(plan.periods).forEach(period => {
              if (period.id === productId) {
                planId = plan.id;
              }
            });
          }
        });

        // ストアの更新
        useStore.getState().updateSubscription({
          id: productId,
          startDate: purchaseDate,
          endDate: expiryDate,
          autoRenew: true,
        });

        useStore.getState().setPlan(planId);

        // クォータの更新
        useStore.getState().updateQuotaLimits();

        return { success: true, plan: planId };
      }

      return { success: false, message: 'Invalid receipt' };
    } catch (error) {
      console.error('Receipt verification failed:', error);
      return { success: false, message: error.message };
    }
  };
  ```

##### 本番モード実装
- `services/subscription.ts` - 本番モード用の決済処理
  ```typescript
  // services/subscription.ts (本番モード部分)
  import * as StoreKit from 'expo-store-kit';
  import * as InAppPurchases from 'expo-in-app-purchases';
  import { Platform } from 'react-native';
  import { supabase } from './supabase';
  import { PLANS } from '../constants/plans';
  import useStore from '../store';

  // プラン購入処理
  export const purchasePlan = async (planId: string, period: 'monthly' | 'halfyearly' | 'yearly') => {
    try {
      // プラン情報取得
      const plan = PLANS[planId];
      if (!plan || !plan.periods || !plan.periods[period]) {
        throw new Error('Invalid plan or period');
      }

      const productId = plan.periods[period].id;

      // プラットフォーム別の購入処理
      if (Platform.OS === 'ios') {
        // iOS: StoreKit 2
        await StoreKit.requestPermissions();
        const products = await StoreKit.getProductsAsync([productId]);

        if (products.length === 0) {
          throw new Error('Product not found');
        }

        const purchase = await StoreKit.purchaseItemAsync(productId);

        if (purchase.resultCode === StoreKit.PurchaseResultCode.SUCCESS) {
          // レシート検証
          return await verifyReceipt(purchase.receipt, 'ios');
        } else {
          throw new Error(`Purchase failed: ${purchase.resultCode}`);
        }
      } else {
        // Android: Google Play Billing
        await InAppPurchases.connectAsync();
        const products = await InAppPurchases.getProductsAsync([productId]);

        if (products.length === 0) {
          throw new Error('Product not found');
        }

        const { responseCode, results } = await InAppPurchases.purchaseItemAsync(productId);

        if (responseCode === InAppPurchases.IAPResponseCode.OK) {
          const purchase = results[0];
          // レシート検証
          return await verifyReceipt(purchase.purchaseToken, 'android');
        } else {
          throw new Error(`Purchase failed: ${responseCode}`);
        }
      }
    } catch (error) {
      console.error('Purchase failed:', error);
      return { success: false, message: error.message };
    }
  };

  // 購入状態の復元
  export const restorePurchases = async () => {
    try {
      let receipt;

      if (Platform.OS === 'ios') {
        // iOS: StoreKit 2
        await StoreKit.requestPermissions();
        const purchases = await StoreKit.getPurchasesAsync();

        if (purchases.length > 0) {
          // 最新の購入を使用
          receipt = purchases[0].receipt;
        }
      } else {
        // Android: Google Play Billing
        await InAppPurchases.connectAsync();
        const { responseCode, results } = await InAppPurchases.getPurchaseHistoryAsync();

        if (responseCode === InAppPurchases.IAPResponseCode.OK && results.length > 0) {
          // 最新の購入を使用
          receipt = results[0].purchaseToken;
        }
      }

      if (receipt) {
        // レシート検証
        return await verifyReceipt(receipt, Platform.OS as 'ios' | 'android');
      }

      return { success: false, message: 'No purchases found' };
    } catch (error) {
      console.error('Restore purchases failed:', error);
      return { success: false, message: error.message };
    }
  };
  ```
